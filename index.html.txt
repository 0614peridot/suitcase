       function calculateAndResizeCanvas() {
           if (!canvas) return;
           const headerH = document.getElementById('main-header').offsetHeight;
           const studioInfoH = 56, studioFooterH = 90, fixedMargin = 32;

       }

       function deleteSelectedObject() {

       async function addToCloset(dataUrl) {
           const img = new Image();
           img.onload = () => {
               const temp = document.createElement('canvas'), maxWidth = 400;
               let w = img.width, h = img.height;
               if (w > maxWidth) { h *= maxWidth / w; w = maxWidth; }
               temp.width = w; temp.height = h;
               temp.getContext('2d').drawImage(img, 0, 0, w, h);
               closet.unshift({ id: Date.now(), src: temp.toDataURL('image/png') });
               saveData(); renderCloset();
           };
           img.src = dataUrl;
       }

       function updateTripDates() {
           const trip = trips.find(t => t.id === selectedTripId);
           const newStart = document.getElementById('edit-trip-start').value;
           const newEnd = document.getElementById('edit-trip-end').value;
           if (!trip || !newStart || !newEnd) return;


           // Update dates
           trip.start = newStart;
           trip.end = newEnd;


           // Cleanup outfits outside new range
           const validDates = getDatesArray(newStart, newEnd);
           const cleanedOutfits = {};
           validDates.forEach(date => {
               if (trip.outfits[date]) cleanedOutfits[date] = trip.outfits[date];
           });
           trip.outfits = cleanedOutfits;


           saveData();
           closeModals();
           selectedDate = validDates[0];
           renderTripList();
           renderTripDetail();
           showToast("Plan Updated!");
       }


       async function handleOutfitDirectUpload(e) {
           const file = e.target.files[0]; if (!file) return;
           const trip = trips.find(t => t.id === selectedTripId); if (!trip || !selectedDate) return;
           const reader = new FileReader();
           reader.onload = async (ev) => {
               const optimized = await optimizeImageForOutfit(ev.target.result);
               trip.outfits[selectedDate] = optimized; saveData(); renderTripDetail(); showToast("Look Uploaded!"); e.target.value = "";
           };
           reader.readAsDataURL(file);
       }


       async function optimizeImageForOutfit(dataUrl) {
           return new Promise((resolve) => {
               const img = new Image();
               img.onload = () => {
                   const temp = document.createElement('canvas'), maxSize = 800;
                   let w = img.width, h = img.height;
                   if (w > h) { if (w > maxSize) { h *= maxSize / w; w = maxSize; } }
                   else { if (h > maxSize) { w *= maxSize / h; h = maxSize; } }
                   temp.width = w; temp.height = h;
                   temp.getContext('2d').drawImage(img, 0, 0, w, h);
                   resolve(temp.toDataURL('image/png', 0.8));
               };
               img.src = dataUrl;
           });
       }
